{{- /* Trinket.io Python embed. Mirrors docs/source/preprocess.py:generate_trinket_iframe so converted lessons render identically to the legacy Sphinx build. Usage: {{< trinket >}} import turtle tina = turtle.Turtle() tina.forward(100) {{< /trinket >}} Params: height optional, pixels. Default = (line_count * 17) + 110. width optional, pixels. Default = 700. type optional Trinket embed type. Default = "python". */ -}} {{- $code := trim .Inner "\n" -}} {{- $lines := len (split $code "\n") -}} {{- $height := .Get "height" | default (printf "%d" (add (mul $lines 17) 110)) -}} {{- /* Width defaults to 100% so the iframe expands with the responsive content area. Authors may pin a fixed pixel width by passing width="600". */ -}} {{- $width := .Get "width" | default "100%" -}} {{- $type := .Get "type" | default "python" -}} {{- /* Match Python's urllib.parse.quote: spaces -> %20, not +. Hugo's urlquery form-encodes spaces as +, which Trinket interprets as literal '+' chars. Replace + with %20 to get path-encoding semantics. */ -}} {{- $encoded := replace ($code | urlquery) "+" "%20" -}} {{- $src := printf "https://trinket.io/tools/1.0/jekyll/embed/%s#code=%s" $type $encoded -}}
